samda Installation
sudo apt-get install samba
sudo smbpasswd -a """root"""
sudo vim /etc/samba/smb.conf
[Insert at the bottom of the code]
[public] the display folder_name
path = /lpr_data #sharing path
available = yes
valid users = root
read only = no
browseable = yes
public = yes
writable = yes
create mask 0777
directory mask 0777

sudo mkdir lpr_data
sudo chmod 777 lpr_data
sudo systemctl restart samba #pi is smba

Modify resolution without real monitor

Section "Monitor"
       Identifier "DSI-0"
       Option    "Ignore"
    EndSection
    
Section "Screen"
   Identifier    "Default Screen"
   Monitor        "Configured Monitor"
   Device        "Default Device"
   SubSection "Display"
       Depth    24
       Virtual 1920 1080
   EndSubSection
EndSection
